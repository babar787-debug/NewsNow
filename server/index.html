const express = require('express'); const app = express(); const PORT = 5000; app.use(express.static('public')); app.get('/api/news', (req, res) => { res.json([ { title: "News Headline 1", description: "Description for news 1" }, { title: "News Headline 2", description: "Description for news 2" } ]); }); app.listen(PORT, () => console.log(`Server running on http://localhost:${PORT}`));
